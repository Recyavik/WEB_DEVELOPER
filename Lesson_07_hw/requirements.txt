# ─── Зависимости проекта ───────────────────────────────────────────────────
# Этот файл содержит список библиотек, которые нужно установить командой:
#   pip install -r requirements.txt
# Docker делает это автоматически при сборке образа.

# FastAPI — веб-фреймворк для создания API и веб-приложений на Python.
# Версия точно зафиксирована (==), чтобы проект работал одинаково у всех.
fastapi==0.135.3

# Uvicorn — сервер (ASGI), который запускает FastAPI-приложение.
# [standard] добавляет поддержку автоматической перезагрузки (--reload)
# и более быструю обработку соединений через дополнительные пакеты.
uvicorn[standard]==0.44.0

# Jinja2 — шаблонизатор: превращает HTML-шаблоны с переменными
# в готовые HTML-страницы, которые видит пользователь в браузере.
jinja2==3.1.6

# python-multipart — нужен для того, чтобы FastAPI умел читать данные
# из HTML-форм (тег <form method="post">). Без него Form() не работает.
python-multipart==0.0.26

# SQLAlchemy — ORM (Object-Relational Mapper): позволяет работать
# с базой данных через Python-классы, а не писать SQL вручную.
sqlalchemy==2.0.49

# Alembic — инструмент миграций для SQLAlchemy.
# Отслеживает изменения схемы БД и применяет их по одному шагу.
alembic==1.18.4

# psycopg2-binary — драйвер (переходник) между Python и PostgreSQL.
# SQLAlchemy сам по себе не умеет общаться с PostgreSQL —
# он использует этот пакет как посредника.
# «binary» означает предсобранный: не требует компилятора на машине.
psycopg2-binary==2.9.11
